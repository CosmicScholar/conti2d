479  int number_x=grdhead.cols;
480  int number_y=grdhead.rows;
481  int N = number_x*number_y;
482  double* originaldata = inputdata;
483  fftw_complex* out_c = (fftw_complex*)fftw_malloc(sizeof(fftw_complex)* N);
484  FFT2d(number_y, number_x, originaldata, out_c); //傅里叶变换
485  double *bounds = grdhead.bounds;
486  double dy = fabs(bounds[3] - bounds[2]) / (number_y - 1), dx = fabs(bounds[1] - bounds[0]) / (number_x - 1);
487  double en = dy*(number_y), em = dx*(number_x);
488  double u, v;
489  double factor = -tp*rph;
490  double factor_conti;
491  int number_x_half = (int)(number_x / 2);
492  int number_y_half = (int)(number_y / 2);
493  int index_ij;
494

496  {
497  v = i / en;
498  for (int j = 0; j <= number_x_half; j++)
499  {
500  u = j / em;
501  index_ij = i*number_x + j;
502  factor_conti = exp(factor*sqrt(u*u + v*v));
503  out_c[index_ij][0] = out_c[index_ij][0] * factor_conti;
504  out_c[index_ij][1] = out_c[index_ij][1] * factor_conti;
505  }
